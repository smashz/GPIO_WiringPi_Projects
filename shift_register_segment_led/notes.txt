#define   SDI   0(wiringpi equivalent pin from gpio)   // (DS) (connected to ds which is serial data pin from hc595)
#define   RCLK  1   // (STCP)
#define   SRCLK 2   // (SHCP)


void hc595_shift(unsigned char dat){
    int i;
    for(i=0;i<8;i++){
        digitalWrite(SDI, 0x80 & (dat << i)); //checks MSB bit7 (first bit)
        digitalWrite(SRCLK, 1);
        delay(1);
        digitalWrite(SRCLK, 0);
    }
        digitalWrite(RCLK, 1);
        delay(1);
        digitalWrite(RCLK, 0);
}

SDI = shift bits into h2595 one at a time ("Serial" data input) (DP)
SRCLK = shifts bits into shift memory register (pins Q0-Q7) (SHCP)
RCLK = latches bits from shift memory register and outputs at once

// basicaly checks first digit of every value to see if its 0 or 1

(0x80 = 1000 0000 0x5b = 11111100 = 1000 0000)
